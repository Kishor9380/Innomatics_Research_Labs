        "def smart_parking_system():\n",
        "    # Take parking capacity as input\n",
        "    capacity = int(input(\"Enter Parking Capacity: \"))\n",
        "\n",
        "    # Take vehicle logs as input (comma separated)\n",
        "    logs_input = input(\"Enter Vehicle Logs (comma separated IN/OUT): \")\n",
        "\n",
        "    # Convert input string into list\n",
        "    vehicle_logs = logs_input.split(\",\")\n",
        "\n",
        "    # Remove extra spaces and convert to uppercase\n",
        "    vehicle_logs = [log.strip().upper() for log in vehicle_logs]\n",
        "\n",
        "    # Tr kiack current parked vehicles\n",
        "    current_parked = 0\n",
        "\n",
        "    # Track peak parking usage\n",
        "    peak_usage = 0\n",
        "\n",
        "    # Loop through logs\n",
        "    for log in vehicle_logs:\n",
        "\n",
        "        # If vehicle enters, increase count\n",
        "        if log == \"IN\":\n",
        "            current_parked += 1\n",
        "\n",
        "        # If vehicle exits, decrease count (not below 0)\n",
        "        elif log == \"OUT\":\n",
        "            if current_parked > 0:\n",
        "                current_parked -= 1\n",
        "\n",
        "        # Update peak usage\n",
        "        if current_parked > peak_usage:\n",
        "            peak_usage = current_parked\n",
        "\n",
        "    # Print results\n",
        "    print(\"Currently Parked Vehicles:\", current_parked)\n",
        "    print(\"Peak Parking Usage:\", peak_usage)\n",
        "\n",
        "    # Check parking status\n",
        "    if current_parked > capacity:\n",
        "        print(\"Parking Status: ALERT! Capacity Exceeded!\")\n",
        "    else:\n",
        "        print(\"Parking Status: Available.\")\n",
        "\n",
        "\n",
        "# Call function\n",
        "smart_parking_system()"

        "def event_registration_controller():\n",
        "    # Take event capacity input\n",
        "    capacity = int(input(\"Enter Event Capacity: \"))\n",
        "\n",
        "    # Take number of registrations\n",
        "    registrations = int(input(\"Enter Number of Registrations: \"))\n",
        "\n",
        "    # Initialize confirmed and waitlist counts\n",
        "    confirmed = 0\n",
        "    waitlisted = 0\n",
        "\n",
        "    # Loop through each registration\n",
        "    for i in range(registrations):\n",
        "        # If capacity not full, confirm registration\n",
        "        if confirmed < capacity:\n",
        "            confirmed += 1\n",
        "        # If capacity full, add to waitlist\n",
        "        else:\n",
        "            waitlisted += 1\n",
        "\n",
        "    # Display results\n",
        "    print(\"Confirmed Registrations:\", confirmed)\n",
        "    print(\"Waitlisted Users:\", waitlisted)\n",
        "\n",
        "    # Check registration status\n",
        "    if registrations >= capacity:\n",
        "        print(\"Registration Status: Closed\")\n",
        "    else:\n",
        "        print(\"Registration Status: Open\")\n",
        "\n",
        "\n",
        "# Call function\n",
